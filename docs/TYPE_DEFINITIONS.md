## Type Guards

```typescript
// Type guard functions for runtime type checking

function isNote(obj: any): obj is Note {
  return obj && 
    typeof obj.id === 'number' &&
    typeof obj.content === 'string' &&
    obj.noteDate instanceof Date &&
    typeof obj.chatId === 'bigint';
}

function isTodo(obj: any): obj is Todo {
  return obj &&
    typeof obj.id === 'number' &&
    typeof obj.key === 'string' &&
    typeof obj.content === 'string' &&
    Array.isArray(obj.tags) &&
    Array.isArray(obj.contexts) &&
    Array.isArray(obj.projects);
}

function isQuestion(obj: any): obj is Question {
  return obj &&
    typeof obj.id === 'number' &&
    typeof obj.chatId === 'bigint' &&
    typeof obj.questionText === 'string' &&
    ['text', 'number', 'binary'].includes(obj.type);
}

function isTaskStatus(status: any): status is TaskStatus {
  return ['new', 'in-progress', 'done', 'canceled', 'snoozed', 'started'].includes(status);
}

function isQuestionType(type: any): type is QuestionType {
  return ['text', 'number', 'binary'].includes(type);
}
```

## Generic Utility Types

```typescript
// Utility types for common patterns

type Optional<T, K extends keyof T> = Omit<T, K> & Partial<Pick<T, K>>;

type RequiredFields<T, K extends keyof T> = T & Required<Pick<T, K>>;

type DeepPartial<T> = {
  [P in keyof T]?: T[P] extends object ? DeepPartial<T[P]> : T[P];
};

type NonNullable<T> = T extends null | undefined ? never : T;

type AsyncReturnType<T extends (...args: any) => any> = T extends (...args: any) => Promise<infer U> ? U : T extends (...args: any) => infer U ? U : any;

// Database relation types
type WithRelations<T, R extends keyof T> = T & {
  [K in R]: NonNullable<T[K]>;
};

// Service method types
type ServiceMethod<T extends (...args: any[]) => any> = T extends (...args: infer P) => Promise<infer R> 
  ? (...args: P) => Promise<R>
  : T;
```
